高斯牛顿法求解拟合问题

1.1问题描述：

针对这样一个问题，

$$
y = exp(ax^2+bx+c)+w
\
w \longrightarrow (0,\sigma^2) \tag{1}
$$

估计参数a,b,c，已经知道的是点对（x，y）

1.2代码如下：

//代码基于Clion，C++14
#include <iostream>
#include <chrono>
#include <opencv2/opencv.hpp>
#include <Eigen/Core>
#include <Eigen/Dense>
using namespace std;
using namespace Eigen;

int main() {
    double ar = 1.0,br = 2.0,cr = 1.0;//真值
    double ae = 2.0,be = -1.0,ce = 5.0;//初值
    
    int N =100;//生成噪声的点数
    double w_sigma = 1.0;//高斯白噪声
    double inv_sigma = 1.0/w_sigma;
    cv::RNG rng;//随机数的对象
    vector<double> x_data,y_data;
    for (int i = 0; i < N; i++) {
        double x = i/100.0;
        x_data.push_back(x);
        y_data.push_back(exp(ar * x * x + br * x + cr) + rng.gaussian(w_sigma * w_sigma)); 
    }
    //保存x_data,y_data数据,用作画图
//    ofstream f("/home/wpf/SLAM/review_slam2/CPP_test/chapter0/optimization/xy1.txt",ios::app);
//    for (int i = 0; i < N; ++i) {
//        f<<x_data[i]<<" ";
//    }
//    cout<<endl;
//    for (int i = 0; i < N; ++i) {
//        f<<y_data[i]<<" ";
//    }

     int iteration = 100;//迭代次数
    double cost = 0,lastcost = 0;
    chrono::steady_clock::time_point t1 = chrono::steady_clock::now();
    for (int iter = 0; iter < iteration; iter++) {
        Eigen::Matrix3d H = Eigen::Matrix3d ::Zero();
        Eigen::Vector3d b = Eigen::Vector3d ::Zero();
        cost = 0;
		//开始迭代
        for (int i = 0; i < N; i++) {
            double xi =x_data[i],yi =y_data[i];
            double error = yi - exp(ae*xi*xi+be*xi+ce);
            Eigen::Vector3d J;
            J[0] = -xi*xi*exp(ae*xi*xi+be*xi+ce);
            J[1] = -xi*exp(ae*xi*xi+be*xi+ce);
            J[2] = -exp(ae*xi*xi+be*xi+ce);
            //顺序不能变
            H += inv_sigma*inv_sigma*J*J.transpose();
            //数乘矩阵等于矩阵乘数
            b += -inv_sigma*inv_sigma*error*J;///kejioahuan
            cost += error*error;
        }
        //Eigen的稠密头文件中包含的算法
        Eigen::Vector3d dx = H.ldlt().solve(b);
        if(isnan(dx[0])){
            cout<<"result is nan!"<<endl;
            break;
        }
        //迭代终止条件，当误差一直减小的时候
        if (iter >0 && cost >= lastcost){
            cout<<"cost :"<<cost<<">= last cost :"<<lastcost<<",break."<<endl;
            break;
        }
		//误差量的更新
        ae += dx[0];
        be += dx[1];
        ce += dx[2];
        lastcost =cost;
        cout<<"total cost :"<<cost<<", \tupdate :"<<dx.transpose()<<",\t estimate:"<<ae<<","<<be<<","<<ce<<endl;
    }
    chrono::steady_clock::time_point t2 = chrono::steady_clock::now();
    chrono::duration<double> time_used = chrono::duration_cast<chrono::duration<double>>(t2-t1);
    cout<<"solve time is :" << time_used.count()<<"seconds"<<endl;
    cout<<"ESTIMATE A,B,C:"<<ae<<","<<be<<","<<ce<<endl;
    std::cout << "Hello, World!" << std::endl;
    return 0;
}
